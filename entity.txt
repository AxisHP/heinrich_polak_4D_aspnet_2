UserEntity {
	Guid PublicId { get; set; } // Public unique identifier (safe to expose externally)
	int Id { get; set; } // Internal database primary key (auto-incremented)
	string FirstName { get; set; } // Given name of the user
	string LastName { get; set; } // Family name / surname of the user
	DateTime DateOfBirth { get; set; } // User's date of birth (for age checks)
	string Email { get; set; } // Contact / login email address
	string PhoneNumber { get; set; } // Contact phone number
	string Address { get; set; } // Postal or billing address
	string Role { get; set; } // Authorization role (e.g., admin, customer)
}

ItemEntity {
	Guid ItemId { get; set; } // Public unique identifier for the item
	int Id { get; set; } // Internal database primary key for the item
	string Name { get; set; } // Display name of the item
	string Description { get; set; } // Detailed item description
	decimal Price { get; set; } // Unit price for the item
	int StockQuantity { get; set; } // Available stock quantity
}

CategoryEntity {
	Guid CategoryId { get; set; } // Public unique identifier for the category
	int Id { get; set; } // Internal database primary key for the category
	string Name { get; set; } // Category display name
	string Description { get; set; } // Optional description of the category
}

OrderEntity {
	Guid OrderId { get; set; } // Public unique identifier for the order
	int Id { get; set; } // Internal database primary key for the order
	Guid UserId { get; set; } // Reference to the placing user's PublicId
	List<OrderItem> Items { get; set; } // Line items included in the order
	decimal TotalAmount { get; set; } // Total amount for the order
	DateTime OrderDate { get; set; } // When the order was placed
	string Status { get; set; } // Current order state (in cart, getting ready, delivering, delivered, finished, canceled)
}

OrderItem {
	Guid ItemId { get; set; } // Public unique identifier of the referenced item
	int Id { get; set; } // Internal database primary key for the order item line
	int Quantity { get; set; } // Quantity of this item in the order
	decimal Price { get; set; } // Price per unit at time of order
}
